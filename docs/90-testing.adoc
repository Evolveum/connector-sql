[[unit-tests-with-h2]]
== Unit Tests with In-Memory Database (H2)

Unit tests use H2 in-memory database for fast, isolated testing without external dependencies.

[source,groovy]
----
import org.testng.annotations.Test

class SqlConnectorUnitTest {

    @Test
    void testSchemaAutoDiscovery() {
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
        config.username = "sa"
        config.password = ""
        config.autoDiscoverSchema = true
        
        connector.initialize(config)
        
        def schema = connector.getSchema()
        assert schema != null
        assert schema.getObjectClass("User") != null
    }
    
    @Test
    void testCrudOperations() {
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
        config.username = "sa"
        config.password = ""
        config.autoDiscoverSchema = true
        
        connector.initialize(config)
        
        // Create
        def user = new ConnectorObject("User")
        user.setUid(new Uid("user1"))
        user.addAttribute("username", "testuser")
        user.addAttribute("email", "test@example.com")
        connector.create(user, null)
        
        // Read
        def retrieved = connector.retrieve("User", new Uid("user1"), null)
        assert retrieved != null
        assert retrieved.getAttributeByName("username").getValue() == "testuser"
        
        // Update
        retrieved.addAttribute("email", "updated@example.com")
        connector.update(new Uid("user1"), 
            Arrays.asList(
                new AttributeBuilder().setName("email").setValues("updated@example.com").build()
            ),
            null
        )
        
        // Delete
        connector.delete(new Uid("user1"), null)
        def deleted = connector.retrieve("User", new Uid("user1"), null)
        assert deleted == null
    }
}

[[integration-tests]]
== Integration Tests with Real Databases

Integration tests verify connector behavior against actual database systems.

[source,groovy]
----
import org.testng.annotations.Test
import org.testng.annotations.AfterClass
import org.testng.annotations.BeforeClass

class SqlConnectorIntegrationTest {

    private SqlConnector connector
    private PostgresContainer<?> postgresContainer

    @BeforeClass
    void setup() {
        postgresContainer = new PostgresContainer<>("postgres:15")
            .withDatabaseName("testdb")
            .withUsername("testuser")
            .withPassword("testpass")
        postgresContainer.start()
        
        connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.jdbcUrl = postgresContainer.getJdbcUrl()
        config.username = postgresContainer.getUsername()
        config.password = postgresContainer.getPassword()
        config.autoDiscoverSchema = true
        
        connector.initialize(config)
    }

    @AfterClass
    void tearDown() {
        if (connector != null) {
            try {
                connector.close()
            } catch (Exception e) {
                // Ignore
            }
        }
        if (postgresContainer != null) {
            postgresContainer.stop()
        }
    }

    @Test
    void testFilterQueries() {
        // Insert test data
        def testData = [
            [username: "alice", email: "alice@test.com", active: true],
            [username: "bob", email: "bob@test.com", active: false],
            [username: "charlie", email: "charlie@test.com", active: true]
        ]
        
        testData.each { row ->
            connector.executeVoid("INSERT INTO users (username, email, active) VALUES (?, ?, ?)",
                row.username, row.email, row.active)
        }
        
        // Test filter with equals
        def filterEquals = new FilterBuilder().isEqualTo("username", "alice").build()
        def results = connector.search("User", filterEquals, null)
        assert results.size() == 1
        assert results[0].getAttributeByName("username").getValue() == "alice"
        
        // Test filter with comparison
        def filterGt = new FilterBuilder().isGreaterThan("id", 0L).build()
        results = connector.search("User", filterGt, null)
        assert results.size() == 3
        
        // Test filter with logical AND
        def filterAnd = new FilterBuilder()
            .and(
                new FilterBuilder().isEqualTo("active", true).build(),
                new FilterBuilder().isLike("username", "%al%").build()
            ).build()
        results = connector.search("User", filterAnd, null)
        assert results.size() == 1
    }

    @Test
    void testTransactionHandling() {
        // Test rollback on error
        connector.executeVoid("INSERT INTO users (username) VALUES (?)", "transaction_test")
        
        try {
            connector.executeVoid("INSERT INTO users (username) VALUES (?)", "will_fail")
            // Intentional error to test rollback
            connector.executeVoid("INSERT INTO invalid_table VALUES (1)")
        } catch (Exception e) {
            // Expected
        }
        
        // Verify transaction was rolled back
        def count = connector.executeQuery("SELECT COUNT(*) FROM users WHERE username = ?",
            "will_fail").executeLongQuery()
        assert count == 0
    }
}

[[mocking-strategies]]
== Mocking Strategies for External Dependencies

Mock external services and dependencies to isolate connector logic.

[source,groovy]
----
import org.mockito.Mockito
import spock.lang.Specification
import spock.lang.Shared

class SqlConnectorMockingSpec extends Specification {

    @Shared
    def mockConnection = Mockito.mock(Connection)
    
    @Shared
    def mockDataSource = Mockito.mock(DataSource)
    
    def setup() {
        Mockito.when(mockDataSource.getConnection()).thenReturn(mockConnection)
    }

    def "test with mocked connection"() {
        given:
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.dataSource = mockDataSource
        config.autoDiscoverSchema = false
        
        when:
        connector.initialize(config)
        
        then:
        connector != null
    }

    def "test error handling with mocked dependencies"() {
        given:
        Mockito.when(mockConnection.prepareStatement(Mockito.anyString()))
            .thenThrow(new SQLException("Connection failed"))
        
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.dataSource = mockDataSource
        
        when:
        connector.initialize(config)
        
        then:
        def exception = thrown(ConfigurationException)
        exception.message.contains("Connection")
    }
}

import org.testng.annotations.Test
import java.sql.Connection

class SqlConnectorTransactionTest {

    @Test
    void testAutoCommitEnabled() {
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
        config.username = "sa"
        config.password = ""
        config.autoCommit = true
        
        connector.initialize(config)
        
        def conn = connector.getConnection()
        assert conn.getAutoCommit() == true
        
        conn.close()
    }

    @Test
    void testAutoCommitDisabled() {
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
        config.username = "sa"
        config.password = ""
        config.autoCommit = false
        
        connector.initialize(config)
        
        def conn = connector.getConnection()
        assert conn.getAutoCommit() == false
        
        conn.close()
    }

    @Test
    void testRollbackOnException() {
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
        config.username = "sa"
        config.password = ""
        config.autoCommit = false
        
        connector.initialize(config)
        
        // Insert initial data
        connector.executeVoid("INSERT INTO users (id, username) VALUES (?, ?)", 1, "initial")
        
        def conn = connector.getConnection()
        
        try {
            conn.setAutoCommit(false)
            
            // Partial operations
            connector.executeVoid(conn, "INSERT INTO users (id, username) VALUES (?, ?)", 2, "test1")
            connector.executeVoid(conn, "INSERT INTO users (id, username) VALUES (?, ?)", 3, "test2")
            
            // Force exception with invalid table
            try {
                connector.executeVoid(conn, "INSERT INTO nonexistent (id) VALUES (1)")
            } catch (Exception e) {
                // Expected
            }
            
            conn.rollback()
        } finally {
            if (conn != null && !conn.getAutoCommit()) {
                conn.setAutoCommit(true)
            }
            conn.close()
        }
        
        // Verify rollback - only initial data should exist
        def results = connector.executeQuery("SELECT COUNT(*) FROM users").executeLongQuery()
        assert results == 1
    }

    @Test
    void testExplicitCommit() {
        def connector = new SqlConnector()
        def config = new SqlConnectorConfiguration()
        config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
        config.username = "sa"
        config.password = ""
        config.autoCommit = false
        
        connector.initialize(config)
        
        def conn = connector.getConnection()
        
        try {
            conn.setAutoCommit(false)
            
            connector.executeVoid(conn, "INSERT INTO users (id, username) VALUES (?, ?)", 1, "committed")
            
            conn.commit()
        } finally {
            conn.close()
        }
        
        // Verify data was committed
        def results = connector.executeQuery("SELECT COUNT(*) FROM users WHERE username = ?",
            "committed").executeLongQuery()
        assert results == 1
    }
}

== Debugging Tips

=== Enabling SQL Logging

Configure logging to debug query execution:

[source,groovy]
----
def config = new SqlConnectorConfiguration()
config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
config.username = "sa"
config.password = ""
config.logSql = true
config.logLevel = "DEBUG"
----

Or use SLF4J directly:

[source,groovy]
----
import org.slf4j.Logger
import org.slf4j.LoggerFactory

Logger logger = LoggerFactory.getLogger(SqlConnector)
logger.debug("Executing query: {}", sql)
----

=== Understanding Query Execution Order

1. Configuration validation and property parsing
2. Connection pool initialization
3. Schema auto-discovery (if enabled)
4. Object class mapping
5. Attribute mapping
6. Query generation from filters
7. Parameter binding
8. Query execution
9. Result mapping
10. Resource cleanup

=== Interpreting Test Failures

Common failure patterns:

* **Connection refused**: Check JDBC URL, credentials, network
* **Table not found**: Verify table exists or auto-discover is enabled
* **Column not found**: Check attribute名称 matches column names
* **Duplicate key**: Ensure unique identifiers or disable auto-increment conflict
* **Timeout**: Increase timeout values or optimize queries

=== Common Issues and Solutions

Issue: "Table USERS not found"
^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
Solution: Enable `autoDiscoverSchema = true` or manually create tables before tests.

Issue: "Column username not found"
^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
Solution: Verify column name matches in database schema or use attribute mapping.

Issue: "Connection pool exhausted"
^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
Solution: Increase pool size or ensure connections are properly closed in finally blocks.

Issue: "Transaction rollback"
^^^^^^^^^^^^^^^^^^^^^^^^^^^^
Solution: Check for exception handlers that may be triggering rollback, verify autoCommit settings.

Issue: "Query timeout exceeded"
^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
Solution: Increase `socketTimeout` configuration or optimize slow queries.
