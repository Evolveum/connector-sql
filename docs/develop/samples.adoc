== H2 in-memory sample

Simple sample using H2 in-memory database for testing.

=== Schema

[source,groovy]
----
objectClass("User") {
    table("users")
    
    attribute("id") {
        sql { type INTEGER; primaryKey true }
    }
    
    attribute("username") {
        sql { type VARCHAR(50); nullable false }
        creatable true
        updatable true
    }
    
    attribute("email") {
        sql { type VARCHAR(255); nullable false }
        creatable true
        updatable true
    }
    
    attribute("is_active") {
        sql { type BOOLEAN; defaultValue "true" }
        creatable true
        updatable true
    }
    
    attribute("created_at") {
        sql { type TIMESTAMP; defaultValue "CURRENT_TIMESTAMP" }
        creatable false
        updatable false
    }
}
----

=== Test data setup

[source,groovy]
----
// Initialize
def config = new SqlConnectorConfiguration()
config.jdbcUrl = "jdbc:h2:mem:testdb;DB_CLOSE_DELAY=-1"
config.username = "sa"
config.password = ""
config.autoDiscoverSchema = true

def connector = new SqlConnector()
connector.initialize(config)

// Create table
connector.executeVoid("""
    CREATE TABLE IF NOT EXISTS users (
        id IDENTITY PRIMARY KEY,
        username VARCHAR(50) UNIQUE NOT NULL,
        email VARCHAR(255) NOT NULL,
        is_active BOOLEAN DEFAULT TRUE,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
""")
----

=== CRUD operations

[source,groovy]
----
// Create
def user = new ConnectorObject("User")
user.addAttribute("username", "john")
user.addAttribute("email", "john@example.com")
connector.create(user, null)

// Read
def retrieved = connector.retrieve("User", user.uid, null)
assert retrieved.getAttributeByName("username").value == "john"

// Update
retrieved.addAttribute("email", "john.doe@example.com")
connector.update(user.uid, retrieved.attributes, null)

// Search
def filter = new FilterBuilder().isEqualTo("is_active", true).build()
def results = connector.search("User", filter, null)
assert results.size() == 1

// Delete
connector.delete(user.uid, null)
----

=== Custom search with full-text

[source,groovy]
----
objectClass("User") {
    table("users")
    
    search {
        handler { filter ->
            def sql = """
                SELECT id, username, email, created_at
                FROM users
                WHERE to_tsvector('english', username || ' ' || email) @@ to_tsquery(?)
            """
            return executeQuery(sql, [filter])
        }
    }
}
----

=== Custom create with sequence

[source,groovy]
----
objectClass("User") {
    table("users")
    
    create {
        handler { attributes ->
            def sql = """
                INSERT INTO users (id, username, email) 
                VALUES (users_seq.NEXTVAL, ?, ?)
                RETURNING id INTO ?
            """
            def result = executeInsert(sql, [
                attributes.username,
                attributes.email
            ])
            return result[0]
        }
    }
}
----

    user.addAttribute("email", "test@example.com")

    assert user.getUid().value != null
}

== Best practices

1. **Start with defaults** - Use auto-generated handlers first
2. **Custom only when needed** - Keep custom logic minimal
3. **Test with multiple databases** - Verify cross-database compatibility
4. **Document schema** - Add descriptions to attributes
5. **Use transactions** - For multi-table operations
